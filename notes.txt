# Docker Compose Microsoft SQL Server ( docker-compose-microsoft.yml)
services:
  sql-server:
    container_name: mssql
    image: mcr.microsoft.com/mssql/server
    hostname: MicrosoftSQL
    ports:
      - "1433:1433"
    environment:
      - ACCEPT_EULA=Y
      - SA_PASSWORD=secret
    volumes:
      - ./data/mssql:/var/opt/mssql/data
    networks:
      - backend

networks:
  backend:
    driver: bridge

# START: COMPOSE_PROJECT_NAME=microsoft docker-compose -f docker-compose-microsoft.yml up -d
